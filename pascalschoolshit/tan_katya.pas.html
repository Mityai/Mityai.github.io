uses
CRT, GRAPH;
const h=0.013;
left=-13;
right=13;
var
 gd,gm:integer;
 x,a,b,k,y,c:real;
 procedure decart;
 begin
  line (10,240,630,240);
  line (320,470,320,10);
  line (310,30,320,10);
  line (330,30,320,10);
  line (610,230,630,240);
  line (610,250,630,240);
  outtextxy (310,5,'Y');
  outtextxy (310,245,'O');
  outtextxy (630,250, 'X');
 end;
 function f1(x:real):real;
 begin
  f1:=a*x*x+b*x+c;
 end;
 function f2(x:real):real;
 begin
  f2:=x*x*x;
 end;
 function f3(x:real):real;
 begin
  f3:=1/x;
 end;
 function f4(x:real):real;
 begin
  f4:=sqrt(x);
 end;
 function f5(x:real):real;
 begin
  f5:=sin(x);
 end;
 function f6(x:real):real;
 begin
  f6:=cos(x);
 end;
 function f7(x:real):real;
 begin
  f7:=sin(x)/cos(x);
 end;
 begin
 k:=25;
 gd:=detect;
 InitGraph (gd,gm,'D:\BP\BGI');
 decart;
 read (a,b,c);
 x:=left;
 y:=f7(x);
 moveto(round(k*x)+320, (-round(k*y)+240));
 setcolor(2);
 while x<=right do
  begin
  y:=f7(x);
  lineto (round(k*x)+320, (-round(k*y)+240));
  if (f7(x) * f7(x+h) < 0) then
    moveto(round(k*(x+h))+320, -round(k*f7(x+h))+240);
  x:=x+h;
  end;
  repeat
  until KeyPressed;
 end.
