uses graph, crt;

const h = 0.13;

var
  D, m : integer;
  i, k, left : integer;
  mPI, x, koef : real;

function getX(x : integer) : integer;
begin
  getX := x + 320;
end;

function getY(y : integer) : integer;
begin
  getY := -y + 240;
end;

function tan(x : real) : real;
begin
  tan := Sin(x) / Cos(x);
end;

procedure Decart;
begin
  Line(  0, 240, 640, 240);
  Line(625, 235, 640, 240);
  Line(625, 245, 640, 240);
  OutTextXY(620, 220, 'x');

  Line(320,   0, 320, 480);
  Line(320,   0, 315,  15);
  Line(320,   0, 325,  15);
  OutTextXY(300, 5, 'y');
end;

begin
  D := detect;
  InitGraph(D, m, '');
  Decart;

  k := 20;
  mPI := 180;
  koef := mPI / Pi;

  for i := -1 to 1 do
  begin
    left := Round(mPI * i);
    x := -Pi / 2 + 0.01;
    MoveTo(getX(left + round(x * koef)), getY(round(k * tan(x))));
    while (x < Pi / 2) do
    begin
      LineTo(getX(left + round(x * koef)), getY(round(k * tan(x))));
      x := x + h;
    end;
  end;

  ReadKey;
end.
